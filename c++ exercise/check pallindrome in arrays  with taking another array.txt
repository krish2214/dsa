#include<iostream>
#include<climits>
using namespace std;
int main(){
    bool flag = true;
    int orgarr[]={1,2,2,1};
    int arr[]={1,2,2,1};
    int sz = sizeof(arr)/sizeof(int);
    // cout<<sz<<endl;
    int first = 0;
    int last = sz-1;
    for(int i=0; i<sz/2; i++){
        swap(arr[first], arr[last]);
        first++;
        last--;
    }
    for(int i=0; i<sz; i++){
        if(arr[i]!=orgarr[i]){
           flag = false;
           break;
        }
    }
    if(flag==true){
        cout << "Yes, it's an pallindrome";
    }else{
        cout << "No, It's not an pallindrome";
    }
    
    return 0;
}
