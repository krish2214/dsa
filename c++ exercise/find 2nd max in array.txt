#include<iostream>
#include<climits>
using namespace std;
int main(){
    int arr[]={1,2,0,3};
    int first = INT_MIN;
    for(int i=0; i<4; i++){
        if(arr[i]>first){
            first = arr[i];
        }
    }
    int s=0;
    for(int i=0; i<4; i++){
        if(arr[i]<first && arr[i]>s){
            s=arr[i];
        }
    }
    cout << s;
    
}
