#include<iostream>
using namespace std;

void intersection(int arr1[], int arr2[], int sz1, int sz2){
    cout << "Intersection : ";
    for(int i=0; i<sz1; i++){
        for(int j=0; j<sz2; j++){
            if(arr1[i]==arr2[j]){
                cout << arr1[i] << " ";
            }
        }
    }
}

int main(){
    int arr1[]={2, 3, 4, 5, 6};
    int sz1 = 5;
    int arr2[]={4, 5, 7, 6};
    int sz2 = 4;
    intersection(arr1, arr2, sz1, sz2);
    return 0;
}
