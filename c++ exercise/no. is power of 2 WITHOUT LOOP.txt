#include <iostream>
using namespace std;

bool powerof2(int n){
    //if(n <= 0) return false;
    if((n&(n-1))==0){
        return true;
    }else{
        return false;
    }
}

int main(){
    int n;
    cout << "Enter a number : ";
    cin >> n;
    
    if(powerof2(n)){
        cout << n << " is a power of 2" << endl;
    }else{
        cout << n << " is not a power of 2" << endl;
    }
    return 0;
}
