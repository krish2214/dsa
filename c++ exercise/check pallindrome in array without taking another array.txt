#include<iostream>
#include<climits>
using namespace std;
int main(){
    bool flag = true;
    int arr[]={1,2,2,1};
    int sz = sizeof(arr)/sizeof(int);
    int first = 0, last = sz-1;
    for(int i=0; i<sz/2; i++){
        if(arr[first]!=arr[last]){
            last--;
            flag = false;
            break;
        }
    }
    if(flag==true){
        cout << "Yes, it's an pallindrome";
    }else{
        cout << "No, it's not an pallindrome";
    }
    return 0;
}
