#include<iostream>
using namespace std;

void checkPrime(int n){
    bool isPrime = true;
    for(int i=2; i<=n/2; i++){
        if(n%i==0){
            isPrime = false;
            break;
        }
    }
    if(isPrime){
        cout << n << " is a prime no.";
    }else{
        cout << n << " is not a prime no.";
    }
}

 int main(){
     int n;
     cout << "Enter a number : ";
     cin >> n;
     checkPrime(n);
 }
