#include <iostream>
using namespace std;

int decimalTobinary(int n){
    int rem, ans=0, power=1;
    while(n!=0){
        rem = n%2;
        n/=2;
        ans+=rem*power;
        power*=10;
    }
    return ans;
}

int main(){
    int n;
    cout << "Enter a decimal base number : ";
    cin >> n;
    cout << "binary form of " << n << " is " << decimalTobinary(n) << endl;
    return 0;
}
