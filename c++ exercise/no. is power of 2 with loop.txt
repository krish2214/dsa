#include <iostream>
using namespace std;

bool powerof2(int n){
  while(n>1){
      
      if(n%2!=0){
          return false;
      }
      n/=2;
  }
  return true;
}

int main(){
    int n;
    cout << "Enter a number : ";
    cin >> n;
    
    if(powerof2(n)){
        cout << n << " is a power of 2" << endl;
    }else{
        cout << n << " is not a power of 2" << endl;
    }
    return 0;
}
